//write a c program to  print the following pattern
#include <stdio.h>
int main() {
    int alp;
    printf("enter an alphabet:\n");
    scanf("%d",&alp);
    for(int i=1;i<=alp;i++)//number of lines
    {
        for(int j=1;j<=alp;j++)//number of times
        {
            printf("%c",(96+i));
        }
        printf("\n");
}
}
